# =====================================================

# RELOJ

# =====================================================

net "clk" LOC="E3" | IOSTANDARD="LVCMOS33";



# =====================================================

# BOTONES

# reset  → BTNC

# ini    → BTNL (play/pause)

# =====================================================

net "reset" LOC="P17" | IOSTANDARD="LVCMOS33";

net "ini"   LOC="M17" | IOSTANDARD="LVCMOS33";



# =====================================================

# SWITCHES

# bt(0) = cinit  → inicialización aleatoria o en cero

# bt(1) = op     → grano aleatorio o en el centro
# bt(2) = gran  - granos aleatorios(cantidad)

 # =====================================================

net "bt<0>" LOC="J15" | IOSTANDARD="LVCMOS33";

net "bt<1>" LOC="R15" | IOSTANDARD="LVCMOS33";

net "bt<2>" LOC="M13" | IOSTANDARD="LVCMOS33";



# =====================================================
